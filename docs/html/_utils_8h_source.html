103static float FindAngle(const gef::Vector4& a, const gef::Vector4& b)
104{
105 double dot_product = a.x() * b.x() + a.y() * b.y() + a.z() * b.z();
106 double mag_a = std::sqrt(a.x() * a.x() + a.y() * a.y() + a.z() * a.z());
107 double mag_b = std::sqrt(b.x() * b.x() + b.y() * b.y() + b.z() * b.z());
108 double cos_theta = dot_product / (mag_a * mag_b);
109 double theta = std::acos(cos_theta) * 180 / PI;
110 return theta;
111}
